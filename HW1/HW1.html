def create_odd_probabilities(df,bookmaker):
    
    Home = bookmaker + "H"
    Draw = bookmaker + "D"
    Away = bookmaker + "A"
    
    HomeProb = Home + "P"
    DrawProb = Draw + "P"
    AwayProb = Away + "P"
    
    df[HomeProb] = 1/df[Home]
    df[DrawProb] = 1/df[Draw]
    df[AwayProb] = 1/df[Away]

def create_normalized_probabilities(df, bookmaker):
    
    HomeProb = bookmaker + "HP"
    DrawProb = bookmaker + "DP"
    AwayProb = bookmaker + "AP"
    
    TotalProb = bookmaker + "PS"
    
    NormalizedHomeProb = HomeProb + "N"
    NormalizedDrawProb = DrawProb + "N"
    NormalizedAwayProb = AwayProb + "N"
    
    df[TotalProb] = 1/(df[HomeProb] + df[DrawProb] + df[AwayProb])
    
    df[NormalizedHomeProb] = df[HomeProb]*df[TotalProb]
    df[NormalizedDrawProb] = df[DrawProb]*df[TotalProb]
    df[NormalizedAwayProb] = df[AwayProb]*df[TotalProb]

def draw_probability(df,bookmaker,normalized = False):
    
    if normalized:
        Home_Away = bookmaker + "H-AN"
        HomeProb = bookmaker + "HPN"
        DrawProb = bookmaker + "DPN"
        AwayProb = bookmaker + "APN"
        
    else:
        Home_Away = bookmaker + "H-A"
        HomeProb = bookmaker + "HP"
        DrawProb = bookmaker + "DP"
        AwayProb = bookmaker + "AP"

    df[Home_Away] = df[HomeProb] - df[AwayProb]

    plt.scatter(df[Home_Away], df[DrawProb])

def draw_binned_probability(df,bookmaker,normalized = False, binstep = 0.05):
    if normalized:
        string = bookmaker + "H-AN"
    else:
        string = bookmaker + "H-A"

    bins = np.arange(-1,1,binstep)
    bins2 = np.arange((-1+binstep/2),(1+binstep/2),binstep)
    n = np.zeros(len(bins))
    for i in range(len(bins)-1):
        d = df[(df[string] >= bins[i]) & (df[string] < bins[i+1])]
        if len(d) == 0:
            n[i] = None
        else:
            n[i] = len(d[d["FTR"]=="D"])/len(d)
    plt.scatter(bins2,n,c="r")

def get_graph(df,bookmaker,normalized = False ,binstep = 0.05):
    
    create_odd_probabilities(df,bookmaker)
    if normalized:
        create_normalized_probabilities(df,bookmaker)
    draw_probability(df,bookmaker,normalized)
    draw_binned_probability(df,bookmaker,normalized,binstep)
    plt.ylabel('P(draw)')
    plt.xlabel('P(home win) - P(away win)')
    plt.title(bookmaker)
